Online Ordering Flow

Figure 7. Online ordering flow from catalog browse through payment processing and pickup

flowchart TD
    C(["Customer"]) --> BROWSE["Browse Catalog
Search & filter"]
    BROWSE --> CART["Add to Cart"]
    CART --> LGN{"Logged in?"}
    LGN -->|No| LOGIN["Login / Register"] --> LGN
    LGN -->|Yes| PAY{"Payment Method"}

    PAY -->|"Cash on Pickup"| SUB["Submit Order"]
    PAY -->|"GCash"| SUB
    PAY -->|"Loyalty Points"| SUB

    SUB --> V1["Verify prices from DB"]
    V1 --> V2["Check stock availability"]
    V2 --> V3["Expiry enforcement"]
    V3 --> V4["Rx product check"]
    V4 --> V5["Validate points balance"]

    V5 --> TXN["Create order record
Status: Pending"]
    TXN --> ITEMS["Insert order items"]
    ITEMS --> DEDUCT["Deduct stock immediately"]
    DEDUCT --> NOTIF["Notify POS staff"]

    NOTIF --> STAFF(["POS Staff"])
    STAFF --> CONFIRM["Confirmed"]
    CONFIRM --> PREP["Preparing"]
    PREP --> READY["Ready for Pickup"]
    READY --> PICKUP["Mark Picked Up"]

    PICKUP --> SALE["Create sales record"]
    SALE --> AWARD["Award loyalty points
500 peso = 25 pts"]
    AWARD --> DONE(["Order Complete"])

    STAFF -->|"Cancel"| CANCEL["Restore stock
Cancel order"]
