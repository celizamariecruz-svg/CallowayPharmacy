Inventory Management Flow

Figure 5. Inventory flow showing stock tracking, expiry monitoring, and alert generation

flowchart TD
    S(["Staff Login"]) --> AUTH{"Permission
Verified?"}
    AUTH -->|No| X["Access Denied"]
    AUTH -->|Yes| DASH["Inventory Dashboard"]

    DASH --> ADD["Add Product"]
    DASH --> MOV["Stock Movement"]
    DASH --> EXP["Expiry Monitor"]
    DASH --> LOW{"Stock ≤
Reorder Level?"}

    ADD --> INS["Insert product
+ initial stock record"]
    INS --> LOG1["Log activity"]

    MOV --> TYP{"Type?"}
    TYP -->|"IN"| SI["stock += qty"]
    TYP -->|"OUT"| SO["stock -= qty"]
    TYP -->|"ADJUST"| SA["stock = new val"]
    SI --> LOCK["Row lock → Update
products & stock_movements"]
    SO --> LOCK
    SA --> LOCK
    LOCK --> LOG2["Log activity"]

    EXP --> SCAN["Scan expiry dates"]
    SCAN --> CHK{"Status?"}
    CHK -->|"Expired"| BLK["Block sale &
deactivate"]
    CHK -->|"Soon
0-30d"| WRN["Warn at POS"]
    CHK -->|"Valid"| OK["Allow sale"]

    BLK --> ALR["Generate Alert"]
    WRN --> ALR
    LOW -->|Yes| LSA["Low Stock Alert"]
    LSA --> ALR

    ALR --> NTF["Notification Tray"]
    ALR --> CRON["Daily Cron"]
    CRON --> EMAL["Email staff"]
